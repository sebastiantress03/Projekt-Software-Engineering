 11    """

 61def slot_index(uhrzeit_str, zeiten):
 62    """
 63    Liefert den Index einer Uhrzeit innerhalb der Zeitslot-Liste.
 64
 65    Parameter:
 66        - uhrzeit_str (str): Zeit im Format "HH:MM"
 67        - zeiten (list[time]): Liste gültiger Spielzeiten
 68
 69    Beispiel:

 72    Rückgabewert:

 75    Fehlerbehandlung:
 76        - ValueError, wenn Uhrzeit nicht in Liste vorhanden ist

 83def baue_status_lists(schedule, teams):
 84    """
 85    Erstellt Statuslisten für alle Teams über den gesamten Zeitverlauf.
 86
 87    Parameter:
 88        - schedule (list): Liste der Spiel-Dictionaries
 89        - teams (dict): team_id → Teamdaten
 90
 91    Beispiel:
 92        - status_lists, zeiten = baue_status_lists(schedule, teams)
 93
 94    Rückgabewert:
 95        - tuple:
 96            - dict: team_id → Liste von Statusbuchstaben ('S', 'P', 'F')
 97            - list: Liste der verwendeten Zeitpunkte
 98
 99    Fehlerbehandlung:
100        - Keine
101
102    Hinweise:
103        - Spiel = 'S', Pfeifen = 'P', sonst 'F' = Frei
104        - Schiedsrichter werden über Teamnamen gematcht
105    """
106
107    zeiten = extrahiere_zeiten(schedule)
108    status_lists = initialisiere_status_lists(teams, zeiten)
109
110    # Umwandlung der Zeit-Strings in time format
111    #def slot_index(uhrzeit_str):
112     #   t = datetime.strptime(uhrzeit_str, "%H:%M").time()
113     #   return zeiten.index(t)
114
115    for match in schedule:
116        # ignoriert Pausen und Einspielzeiten
117        if 'Match Type' not in match:
118            continue
119
120        # setze den index für einen Timeslot
121        idx = slot_index(match['Uhrzeit'], zeiten)
122
123        # spielende Teams
124        for tid, data in teams.items():
125            if data['name'] in (match['Team 1'], match['Team 2']):
126                status_lists[tid][idx] = 'S'
127
128        # Schiedsrichter
129        ref = match.get('Schiedsrichter')
130        if ref and ref not in ('Not required', None):
131            ref_tid = next(t for t, d in teams.items() if d['name'] == ref)
132            status_lists[ref_tid][idx] = 'P'
133
134    return status_lists, zeiten

443            }
444            team_names.append(name)
445            tid_counter += 1
446
