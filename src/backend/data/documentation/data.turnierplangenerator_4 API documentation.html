  54    total_groups = len(group_names)

  56    base_fields_per_group = fields // total_groups
  57    remaining_fields = fields % total_groups

 454def optimize_schedule(schedule, teams, match_duration, fields, field_assignment):
 455    """
 456    Optimiert den Spielplan hinsichtlich Fairness, Gleichverteilung und Vermeidung von Schiedsrichterkonflikten.

 472        - Bewertet Pläne mit einer Kostenfunktion (`cost()`), u.a. für Leerlaufzeiten, Schiri-Konflikte, ungenutzte Felder.

 481        Konvertiert einen Zeit-String in ein datetime-Objekt.

 494    def build_match_pairs(matches):
 495        """
 496        Sucht passende Hin- und Rückspiel-Paare im Spielplan.
 497
 498        Parameter:
 499            - matches (list): Liste der Spiel-Dictionaries.
 500
 501        Rückgabewert:
 502            - list: Liste von Tupeln mit je einem Hin- und Rückspiel.
 503
 504        Hinweise:
 505            - Nur Matches mit "Hinspiel" werden betrachtet.
 506            - Paarung erfolgt nach Teamnamen und Gruppenabgleich.
 507        """
 508        pairs, used = [], set()
 509        for i, m1 in enumerate(matches):
 510            if i in used or m1.get("Match Type") != "Hinspiel":
 511                continue
 512            for j, m2 in enumerate(matches):
 513                if j in used or j == i:
 514                    continue
 515                if (m2.get("Match Type") == "Rückspiel"
 516                    and m1["Team 1"] == m2["Team 2"]
 517                    and m1["Team 2"] == m2["Team 1"]
 518                    and m1["Gruppe"] == m2["Gruppe"]):
 519                    pairs.append((m1, m2))
 520                    used.update([i, j])
 521                    break

 524    def cost(schedule):
 525        """
 526        Bewertet den Spielplan nach Fairness, Feldnutzung und Schiri-Verteilung.
 527
